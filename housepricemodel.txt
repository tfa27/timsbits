        plt.ylabel('price')

drop_cols = ['date', 'id', 'zipcode']

df

df = df[(df != 0).all(1)]

label = df['price']
feat = df.drop(columns = ['price'], axis = 1)

LinReg = LinearRegression(normalize = True)

LinReg.fit(feat, label)

print(LinReg.score(feat, label))
